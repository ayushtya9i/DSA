Given an array Arr of size N, print second largest distinct element from an array.

class Solution {
    int print2largest(int arr[], int n) {
        // code here
        int l1=Integer.MIN_VALUE,l2=Integer.MIN_VALUE;
        for(int i=0;i<n;i++){
            if(arr[i]>l1){
                l2=l1;
                l1=arr[i];
            }
            if(arr[i]>l2&&arr[i]<l1)
            l2=arr[i];
        }
        if(l2==Integer.MIN_VALUE)
        l2=-1;
        return l2;
    }
}
